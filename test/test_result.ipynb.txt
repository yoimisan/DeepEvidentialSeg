    "def visualize_with_legend(ax, mask, title, label2name):\n",
    "    # 1. Get unique classes present in this specific mask\n",
    "    unique_labels = np.unique(mask)\n",
    "    \n",
    "    # 2. Generate a colormap with enough colors for the max label index (25)\n",
    "    # using 'tab20' or 'jet' usually provides distinct colors\n",
    "    base_cmap = plt.get_cmap('tab20b', 25) \n",
    "    \n",
    "    # 3. Plot the image\n",
    "    im = ax.imshow(mask, cmap=base_cmap, vmin=0, vmax=24)\n",
    "    ax.set_title(title)\n",
    "    \n",
    "    # 4. Create the legend handles manually\n",
    "    legend_patches = []\n",
    "    for label_idx in unique_labels:\n",
    "        if label_idx in label2name:\n",
    "            # Get the color corresponding to this label from the colormap\n",
    "            color = base_cmap(label_idx / 24.0) \n",
    "            patch = mpatches.Patch(color=color, label=f\"{label_idx}: {label2name[label_idx]}\")\n",
    "            legend_patches.append(patch)\n",
    "    \n",
    "    # 5. Add legend to the side of the plot\n",
    "    ax.legend(handles=legend_patches, bbox_to_anchor=(1.05, 1), loc='upper left', borderaxespad=0.)\n"
   ]
